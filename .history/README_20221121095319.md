```
pip install requirements.txt
```

```

   'default': {
       'ENGINE': 'django.db.backends.mysql',
       'NAME':'asc_data',
       'USER': 'root',
       'PASSWORD': '22450438',
       'HOST': '127.0.0.1',
       'POST': '3306'
  }
}

python manage,py createsuperuser #然后需要先创建一个超级用户 在终端会有提示 ，根据提示输入即可
```
